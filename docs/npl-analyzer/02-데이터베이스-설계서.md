# 02. 데이터베이스 설계서 (Supabase PostgreSQL)

## 1. ERD 개요

```
users (1) ─── (N) projects (1) ─── (1) case_info
                    │
                    ├── (1) assumptions
                    ├── (N) auction_rounds
                    ├── (N) rights_claims
                    ├── (N) tenants
                    ├── (N) bulk_properties       [npl_bulk 모드]
                    ├── (N) appraisal_details     [선택]
                    └── (N) analysis_results      [스냅샷]
```

## 2. 테이블 상세 DDL

### 2.1 users — 사용자
```sql
CREATE TABLE public.users (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  email TEXT UNIQUE NOT NULL,
  name TEXT NOT NULL DEFAULT '',
  company TEXT DEFAULT '',
  role TEXT NOT NULL DEFAULT 'investor'
    CHECK (role IN ('investor', 'analyst', 'admin')),
  phone TEXT DEFAULT '',
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- RLS
ALTER TABLE public.users ENABLE ROW LEVEL SECURITY;
CREATE POLICY "users_select_own" ON public.users
  FOR SELECT USING (auth.uid() = id);
CREATE POLICY "users_update_own" ON public.users
  FOR UPDATE USING (auth.uid() = id);

-- Trigger: updated_at 자동 갱신
CREATE OR REPLACE FUNCTION update_updated_at()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = now();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER set_updated_at
  BEFORE UPDATE ON public.users
  FOR EACH ROW EXECUTE FUNCTION update_updated_at();
```

### 2.2 projects — 프로젝트 (분석 단위)
```sql
CREATE TABLE public.projects (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID NOT NULL REFERENCES public.users(id) ON DELETE CASCADE,

  -- 기본 식별
  title TEXT NOT NULL DEFAULT '새 분석',
  analysis_mode TEXT NOT NULL
    CHECK (analysis_mode IN ('auction_single', 'npl_single', 'npl_bulk')),
  status TEXT NOT NULL DEFAULT 'draft'
    CHECK (status IN ('draft', 'in_progress', 'completed', 'archived')),

  -- 메타
  description TEXT DEFAULT '',
  tags TEXT[] DEFAULT '{}',

  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- 인덱스
CREATE INDEX idx_projects_user_id ON public.projects(user_id);
CREATE INDEX idx_projects_status ON public.projects(status);
CREATE INDEX idx_projects_mode ON public.projects(analysis_mode);

-- RLS
ALTER TABLE public.projects ENABLE ROW LEVEL SECURITY;
CREATE POLICY "projects_own" ON public.projects
  FOR ALL USING (auth.uid() = user_id);

CREATE TRIGGER set_projects_updated_at
  BEFORE UPDATE ON public.projects
  FOR EACH ROW EXECUTE FUNCTION update_updated_at();
```

### 2.3 case_info — 사건/물건 기본정보
```sql
CREATE TABLE public.case_info (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  project_id UUID NOT NULL UNIQUE REFERENCES public.projects(id) ON DELETE CASCADE,

  -- ===== 사건 정보 =====
  case_number TEXT DEFAULT '',              -- 예: "2024타경36029"
  court TEXT DEFAULT '',                    -- 예: "대구지방법원 서부지원"
  case_type TEXT DEFAULT '임의경매'
    CHECK (case_type IN ('임의경매', '강제경매', '기타')),
  debtor TEXT DEFAULT '',                   -- 채무자
  owner TEXT DEFAULT '',                    -- 소유자

  -- ===== 날짜 =====
  filing_date DATE,                         -- 경매접수일
  decision_date DATE,                       -- 개시결정일
  appraisal_date DATE,                      -- 감정기일
  claim_deadline DATE,                      -- 배당요구종기
  cancellation_base_date DATE,              -- 말소기준등기일
  next_auction_date DATE,                   -- 다음 매각기일
  construction_date DATE,                   -- 준공일/보존등기일

  -- ===== 물건 기본 =====
  property_type TEXT DEFAULT ''
    CHECK (property_type IN (
      '공장', '상가', '오피스텔', '아파트', '다세대', '다가구',
      '토지', '근린생활시설', '사무실', '주상복합', '기타', ''
    )),
  zoning TEXT DEFAULT '',                   -- 용도지역 (제3종일반주거, 일반상업 등)
  is_regulated BOOLEAN NOT NULL DEFAULT false, -- 규제지역 여부
  address TEXT DEFAULT '',                  -- 소재지
  road_address TEXT DEFAULT '',             -- 도로명주소

  -- ===== 가격 정보 (단위: 원) =====
  appraisal_value BIGINT DEFAULT 0,         -- 감정가 (담보평가액)
  current_minimum_price BIGINT DEFAULT 0,   -- 현재 최저매각가
  ai_estimated_value BIGINT,                -- AI 추정 시세 (NULL 가능)
  official_land_price BIGINT,               -- 공시지가 (원/㎡)

  -- ===== 면적 =====
  land_area_sqm NUMERIC(12,2) DEFAULT 0,   -- 토지면적 (㎡)
  building_area_sqm NUMERIC(12,2) DEFAULT 0,-- 건물면적/전용면적 (㎡)
  land_appraisal BIGINT DEFAULT 0,          -- 토지 감정가
  building_appraisal BIGINT DEFAULT 0,      -- 건물 감정가

  -- ===== 기타 =====
  appraiser TEXT DEFAULT '',                -- 감정평가법인명
  bid_deposit_rate NUMERIC(5,4) NOT NULL DEFAULT 0.1, -- 입찰보증금률 (0.1 or 0.2)
  main_creditor TEXT DEFAULT '',            -- 주채권자
  claim_total BIGINT DEFAULT 0,             -- 청구금액

  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE TRIGGER set_case_info_updated_at
  BEFORE UPDATE ON public.case_info
  FOR EACH ROW EXECUTE FUNCTION update_updated_at();
```

### 2.5 rights_claims — 등기부 권리/채권 현황
```sql
CREATE TABLE public.rights_claims (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  project_id UUID NOT NULL REFERENCES public.projects(id) ON DELETE CASCADE,

  sort_order INT NOT NULL DEFAULT 0,         -- 순번 (표시 순서)
  registration_date DATE,                    -- 접수일

  right_type TEXT NOT NULL DEFAULT '근저당권'
    CHECK (right_type IN (
      '소유권보존', '소유권이전', '근저당권', '저당권',
      '가압류', '압류', '임의경매', '강제경매',
      '전세권', '지상권', '지역권', '임차권', '기타'
    )),

  right_holder TEXT NOT NULL DEFAULT '',     -- 권리자/채권자명

  -- 금액 (단위: 원)
  claim_amount BIGINT NOT NULL DEFAULT 0,    -- 채권금액/최고액
  principal_amount BIGINT DEFAULT 0,         -- 원금 (이자 계산용)
  interest_rate NUMERIC(7,5) DEFAULT 0,      -- 연체이자율 (예: 0.15000)
  interest_start_date DATE,                  -- 연체이자 기산일
  expenses BIGINT DEFAULT 0,                 -- 비용 (채권계산서 기준)

  -- 분석 속성
  priority_rank INT,                         -- 순위 (1순위, 2순위...)
  extinction TEXT NOT NULL DEFAULT '소멸'
    CHECK (extinction IN ('소멸', '존속', '확인필요')),

  -- 분류 (npl_bulk 모드에서 핵심)
  classification TEXT NOT NULL DEFAULT '후순위'
    CHECK (classification IN (
      '선순위', '매입채권(NPL)', '후순위', '가압류/압류', '해당없음'
    )),

  -- 추가 정보
  is_cancellation_base BOOLEAN DEFAULT false, -- 말소기준등기 여부
  is_analysis_target BOOLEAN DEFAULT false,   -- 분석 대상 채권 여부 (npl_single)
  notes TEXT DEFAULT '',

  created_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE INDEX idx_rights_claims_project ON public.rights_claims(project_id);
CREATE INDEX idx_rights_claims_classification ON public.rights_claims(classification);
```

```sql
CREATE TABLE public.tenants (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
```

### 2.7 assumptions — 핵심 가정

```sql
CREATE TABLE public.assumptions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

  auction_execution_cost BIGINT NOT NULL DEFAULT 0,          -- 경매집행비용 (원)

  -- ===== NPL 채권매입 가정 (npl_single, npl_bulk) =====
  -- 할인율 시나리오

  acquisition_tax_rate NUMERIC(7,5) NOT NULL DEFAULT 0.04600,  -- 취득세율 (4.6%)
  other_acquisition_rate NUMERIC(7,5) NOT NULL DEFAULT 0.00500, -- 기타비용률 (0.5%)
  holding_period_years INT NOT NULL DEFAULT 5,                  -- 보유기간 (년)
  annual_appreciation_rate NUMERIC(7,5) NOT NULL DEFAULT 0.02000, -- 연간 자본이득율 (2%)

  operating_expense_rate NUMERIC(5,4) NOT NULL DEFAULT 0.1500, -- 운영비율 (15%)

  renovation_cost BIGINT NOT NULL DEFAULT 50000000,    -- 보수비 (5천만)
```

### 2.8 bulk_properties — 복수 물건 현황 (npl_bulk 모드)
```sql
CREATE TABLE public.bulk_properties (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  project_id UUID NOT NULL REFERENCES public.projects(id) ON DELETE CASCADE,

  property_number INT NOT NULL DEFAULT 1,    -- 물번
  unit_number TEXT DEFAULT '',               -- 호실 (101호 등)
  building_name TEXT DEFAULT '',             -- 건물명
  address TEXT DEFAULT '',                   -- 주소

  property_use TEXT NOT NULL DEFAULT '오피스텔'
    CHECK (property_use IN (
      '오피스텔', '근린상가', '다세대', '아파트',
      '사무실', '토지', '건물', '기타'
    )),
  floor_info TEXT DEFAULT '',                -- 층 정보 (1층, 지하1층 등)

  -- 면적
  land_area_sqm NUMERIC(12,2) DEFAULT 0,    -- 대지권 (㎡)
  exclusive_area_sqm NUMERIC(12,2) DEFAULT 0, -- 전용면적 (㎡)

  -- 가격
  appraisal_value BIGINT NOT NULL DEFAULT 0, -- 감정가 (원)
  fail_count INT NOT NULL DEFAULT 0,         -- 유찰 횟수
  current_minimum BIGINT NOT NULL DEFAULT 0, -- 최저가 (현재)
  assumed_sale_rate NUMERIC(5,4) DEFAULT 0.3400, -- 낙찰가율 (가정)

  -- 개별담보 시 선순위
  individual_senior_claim BIGINT DEFAULT 0,  -- 선순위채권 (개별담보 시)

  sort_order INT NOT NULL DEFAULT 0,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- 계산 필드 (뷰 또는 프론트에서):
-- land_area_pyeong = land_area_sqm * 0.3025
-- exclusive_area_pyeong = exclusive_area_sqm * 0.3025
-- expected_sale_price = appraisal_value * assumed_sale_rate

CREATE INDEX idx_bulk_properties_project ON public.bulk_properties(project_id);
CREATE INDEX idx_bulk_properties_use ON public.bulk_properties(property_use);
```

### 2.10 analysis_results — 분석 결과 스냅샷
```sql
CREATE TABLE public.analysis_results (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  project_id UUID NOT NULL REFERENCES public.projects(id) ON DELETE CASCADE,

  result_type TEXT NOT NULL
    CHECK (result_type IN (
      'dividend_waterfall',   -- 배당 워터폴
      'npl_yield',           -- NPL 수익률
      'auction_irr',         -- 직접낙찰 IRR
      'strategy_comparison', -- 전략 비교
      'sensitivity',         -- 민감도 분석
      'bulk_equity',         -- 복수물건 에쿼티 수익
      'risk_assessment',     -- 리스크 평가
      'summary'             -- 요약 대시보드
    )),

  -- 결과 데이터 전체를 JSONB로 저장
  result_data JSONB NOT NULL DEFAULT '{}'::jsonb,

  -- 메타
  version INT NOT NULL DEFAULT 1,            -- 결과 버전 (재분석 시 증가)
  created_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE INDEX idx_analysis_results_project ON public.analysis_results(project_id);
CREATE INDEX idx_analysis_results_type ON public.analysis_results(result_type);
-- 최신 결과 조회용 복합 인덱스
CREATE INDEX idx_analysis_results_latest
  ON public.analysis_results(project_id, result_type, created_at DESC);
```

## 3. RLS (Row Level Security) 정책

```sql
-- 모든 자식 테이블에 동일한 패턴 적용
-- "프로젝트 소유자만 해당 프로젝트의 데이터에 접근 가능"

-- case_info
ALTER TABLE public.case_info ENABLE ROW LEVEL SECURITY;
CREATE POLICY "case_info_own" ON public.case_info FOR ALL
  USING (project_id IN (SELECT id FROM public.projects WHERE user_id = auth.uid()));

-- auction_rounds
ALTER TABLE public.auction_rounds ENABLE ROW LEVEL SECURITY;
CREATE POLICY "auction_rounds_own" ON public.auction_rounds FOR ALL
  USING (project_id IN (SELECT id FROM public.projects WHERE user_id = auth.uid()));

-- rights_claims
ALTER TABLE public.rights_claims ENABLE ROW LEVEL SECURITY;
```
